```
1. wget https://www.python.org/ftp/python/3.6.3/Python-3.6.3.tar.xz
2. tar -xvf Python-3.6.3.tar.xz /usr/local
3. cd /usr/local/ Python-3.6.3
4. ./configure
5. make
6. make install
7. cd /usr/bin 
8. sudo ln -s /usr/local/bin/python3 python
```
